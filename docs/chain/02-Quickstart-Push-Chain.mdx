import { useEffect } from 'react';
import { useHistory } from '@docusaurus/router';

export default function ChainRedirect() {
  const history = useHistory();
  
  useEffect(() => {
    history.replace('/docs');
  }, [history]);

  return (
    <div>
      <h1>Redirecting...</h1>
      <p>If you are not redirected automatically, <a href="/docs">click here to go to the main docs</a>.</p>
    </div>
  );
}

<ChainRedirect />
